// –°–∏—Å—Ç–µ–º–∞ –ø–æ–¥—Å—á—ë—Ç–∞ –æ—á–∫–æ–≤
class ScoreSystem {
    constructor() {
        this.basePoints = {
            win: 10,
            quickWin: 25,        // ‚â§ 3 –ø–æ–ø—ã—Ç–∫–∏
            perfectGame: 100,    // 1 –ø–æ–ø—ã—Ç–∫–∞
            streakBonus: 5,      // –∑–∞ –∫–∞–∂–¥—É—é –∏–≥—Ä—É –≤ —Å–µ—Ä–∏–∏
            difficultyBonus: 10  // –∑–∞ –∫–∞–∂–¥—ã–π —É—Ä–æ–≤–µ–Ω—å —Å–ª–æ–∂–Ω–æ—Å—Ç–∏
        };
        
        this.difficultyMultipliers = {
            easy: 1.0,
            medium: 1.5,
            hard: 2.0,
            extreme: 3.0
        };
    }
    
    // –í—ã—á–∏—Å–ª—è–µ–º –æ—á–∫–∏ –∑–∞ –∏–≥—Ä—É
    calculateGameScore(gameResult) {
        const { 
            won, 
            attempts, 
            difficulty, 
            maxAttempts, 
            winStreak 
        } = gameResult;
        
        if (!won) return 0; // –ù–µ—Ç –æ—á–∫–æ–≤ –∑–∞ –ø–æ—Ä–∞–∂–µ–Ω–∏–µ
        
        let score = this.basePoints.win;
        
        // –ë–æ–Ω—É—Å –∑–∞ –±—ã—Å—Ç—Ä–æ—Ç—É
        if (attempts === 1) {
            score += this.basePoints.perfectGame;
        } else if (attempts <= 3) {
            score += this.basePoints.quickWin;
        }
        
        // –ë–æ–Ω—É—Å –∑–∞ —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω–æ—Å—Ç—å (–ø—Ä–æ—Ü–µ–Ω—Ç –æ—Ç –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã—Ö –ø–æ–ø—ã—Ç–æ–∫)
        const efficiency = (1 - (attempts / maxAttempts)) * 50;
        score += Math.floor(efficiency);
        
        // –ë–æ–Ω—É—Å –∑–∞ —Å–µ—Ä–∏—é –ø–æ–±–µ–¥
        if (winStreak > 1) {
            score += (winStreak - 1) * this.basePoints.streakBonus;
        }
        
        // –ú–Ω–æ–∂–∏—Ç–µ–ª—å –∑–∞ —Å–ª–æ–∂–Ω–æ—Å—Ç—å
        score *= this.difficultyMultipliers[difficulty];
        
        return Math.floor(score);
    }
    
    // –í—ã—á–∏—Å–ª—è–µ–º –æ—á–∫–∏ –∑–∞ –¥–æ—Å—Ç–∏–∂–µ–Ω–∏–µ
    calculateAchievementScore(achievement) {
        const basePoints = achievement.points || 10;
        const rarity = achievement.rarity || 'common';
        const multiplier = RARITY_CONFIG[rarity].multiplier;
        
        return Math.floor(basePoints * multiplier);
    }
    
    // –§–æ—Ä–º–∞—Ç–∏—Ä—É–µ–º –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏–µ –æ—á–∫–æ–≤
    formatScore(score) {
        if (score >= 1000000) {
            return (score / 1000000).toFixed(1) + 'M';
        } else if (score >= 1000) {
            return (score / 1000).toFixed(1) + 'K';
        }
        return score.toString();
    }
}
